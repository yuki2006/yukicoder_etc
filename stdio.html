C++

#include <iostream>
#include <string>

using namespace std;

int main() {
	int n;
	string s;

	// 標準入力から、空白や改行で区切られた整数と文字列を読み込む。
	cin >> n >> s;

	// 整数と文字列を空白で区切って、標準出力に書き出す。
	cout << n << " " << s << endl;
	return 0;
}
