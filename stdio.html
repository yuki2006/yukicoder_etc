各問題の解き方の例です。 ここでは、No.9000 入出力の練習問題（テスト用）の各問題やり方を例にあげます。

Java

import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        //標準入力から読み込む際に、Scannerオブジェクトを使う。
        Scanner sc = new Scanner(System.in);

        //int １つ分を読み込む
        int a = sc.nextInt();

        //int １つ分を読み込む
        int b = sc.nextInt();

        //string １つ分を読み込む
        String s = sc.next();

        //標準出力に書き出す。
        System.out.println( (a+b) +" "+s);
    }
}

C++

#include <iostream>
#include <string>

using namespace std;

int main() {
	int n;
	string s;

	// 標準入力から、空白や改行で区切られた整数と文字列を読み込む。
	cin >> n >> s;

	// 整数と文字列を空白で区切って、標準出力に書き出す。
	cout << n << " " << s << endl;
	return 0;
}
